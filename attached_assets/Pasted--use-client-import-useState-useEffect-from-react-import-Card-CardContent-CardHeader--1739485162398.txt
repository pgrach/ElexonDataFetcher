"use client"

import { useState, useEffect } from "react"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs"

type SummaryData = {
  energyCurtailed: number
  bitcoinMined: number
  revenue: number
}

export default function SummaryReports() {
  const [summaries, setSummaries] = useState<{
    daily: SummaryData
    monthly: SummaryData
    yearly: SummaryData
  } | null>(null)

  useEffect(() => {
    // TODO: Fetch real summary data from your API
    // For now, we'll use mock data
    const mockSummaries = {
      daily: {
        energyCurtailed: 240, // MWh
        bitcoinMined: 0.05, // BTC
        revenue: 1500, // $
      },
      monthly: {
        energyCurtailed: 7200, // MWh
        bitcoinMined: 1.5, // BTC
        revenue: 45000, // $
      },
      yearly: {
        energyCurtailed: 87600, // MWh
        bitcoinMined: 18, // BTC
        revenue: 540000, // $
      },
    }
    setSummaries(mockSummaries)
  }, [])

  const SummaryCard = ({ data }: { data: SummaryData }) => (
    <div className="grid grid-cols-3 gap-4">
      <Card>
        <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
          <CardTitle className="text-sm font-medium">Energy Curtailed</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold">{data.energyCurtailed} MWh</div>
        </CardContent>
      </Card>
      <Card>
        <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
          <CardTitle className="text-sm font-medium">Bitcoin Mined</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold">{data.bitcoinMined} BTC</div>
        </CardContent>
      </Card>
      <Card>
        <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
          <CardTitle className="text-sm font-medium">Revenue</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold">${data.revenue}</div>
        </CardContent>
      </Card>
    </div>
  )

  return (
    <Card>
      <CardHeader>
        <CardTitle>Summary Reports</CardTitle>
      </CardHeader>
      <CardContent>
        <Tabs defaultValue="daily" className="space-y-4">
          <TabsList>
            <TabsTrigger value="daily">Daily</TabsTrigger>
            <TabsTrigger value="monthly">Monthly</TabsTrigger>
            <TabsTrigger value="yearly">Yearly</TabsTrigger>
          </TabsList>
          <TabsContent value="daily">
            {summaries ? <SummaryCard data={summaries.daily} /> : <p>Loading daily summary...</p>}
          </TabsContent>
          <TabsContent value="monthly">
            {summaries ? <SummaryCard data={summaries.monthly} /> : <p>Loading monthly summary...</p>}
          </TabsContent>
          <TabsContent value="yearly">
            {summaries ? <SummaryCard data={summaries.yearly} /> : <p>Loading yearly summary...</p>}
          </TabsContent>
        </Tabs>
      </CardContent>
    </Card>
  )
}
